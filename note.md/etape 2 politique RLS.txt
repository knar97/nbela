-- Fonction : crée une ligne dans "profils" quand un compte est créé dans auth.users
create function public.gerer_nouvel_utilisateur()
returns trigger
language plpgsql
security definer set search_path = public
as $$
begin
  insert into public.profils (id, nom, prenom, telephone)
  values (
    new.id,
    new.raw_user_meta_data->>'nom',
    new.raw_user_meta_data->>'prenom',
    new.raw_user_meta_data->>'telephone'
  );
  return new;
end;
$$;

-- Déclencheur : exécute la fonction ci-dessus après chaque inscription
create trigger au_nouvel_utilisateur
  after insert on auth.users
  for each row execute function public.gerer_nouvel_utilisateur();
