-- LECTURE : chacun peut lire son propre profil
create policy "Lire son propre profil"
on profils for select
using (auth.uid() = id);

-- CRÉATION : un utilisateur ne crée que son propre profil
create policy "Creer son propre profil"
on profils for insert
with check (auth.uid() = id);

-- MODIFICATION : chacun modifie uniquement son propre profil
create policy "Modifier son propre profil"
on profils for update
using (auth.uid() = id);
